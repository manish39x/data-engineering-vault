## My Solution

```sql
-- Write your PostgreSQL query statement below
WITH reaction_cnt AS (
    SELECT
        user_id,
        reaction,
        COUNT(*) AS cnt
    FROM reactions
    GROUP BY user_id, reaction
), ranked AS (
    SELECT
        user_id,reaction,
        cnt,
        SUM(cnt) OVER(PARTITION BY user_id) AS total_cnt,
        ROW_NUMBER() OVER(PARTITION BY user_id ORDER BY cnt DESC) AS rn
    FROM reaction_cnt
), valid_users AS (
    SELECT user_id
    FROM reactions
    GROUP BY user_id
    HAVING COUNT(DISTINCT content_id) >= 5
)
SELECT
    r.user_id, r.reaction AS dominant_reaction,
    ROUND((r.cnt*1.00 / r.total_cnt), 2) AS reaction_ratio
FROM ranked r
INNER JOIN valid_users v ON v.user_id = r.user_id
WHERE r.rn = 1
AND (r.cnt*1.00 / r.total_cnt) >= 0.6
ORDER BY reaction_ratio DESC
```
